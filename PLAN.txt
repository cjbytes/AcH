```rust
#[derive(Debug, Clone)]
pub struct ChatMessage {
    pub role: String,   // "user" or "assistant"
    pub content: String,

pub struct AiChatAgent {
    history: Arc<Mutex<Vec<ChatMessage>>>,
}

    pub fn new() -> Self {
        Self {
            history: Arc::new(Mutex::new(Vec::new())),
        }
    }

    pub async fn send_message(&self, user_input: &str) -> Result<String, Box<dyn std::error::Error>> {
        // Simulate AI response for now
        let simulated_response = format!("🤖 You said: '{}'. I'm an AI agent!", user_input);

        let mut history = self.history.lock().await;
        history.push(ChatMessage {
            role: "user".to_string(),
            content: user_input.to_string(),
        });
        history.push(ChatMessage {
            role: "assistant".to_string(),
            content: simulated_response.clone(),
        });

        Ok(simulated_response)
    }
```

```rust
use eframe::{egui, Theme};
use egui::TextStyle;
use ach_agent::AiChatAgent;
use tokio;

struct AICHTApp {
    agent: AiChatAgent,
    user_input: String,
    messages: Vec<String>,
}

impl Default for AICHTApp {
    fn default() -> Self {
        Self {
            agent: AiChatAgent::new(),
            user_input: String::new(),
            messages: Vec::new(),
        }
    }
}

impl eframe::App for AICHTApp {
    fn update(&mut self, ctx: &egui::Context, _frame: &mut eframe::Frame) {
        egui::CentralPanel::default().show(ctx, |ui| {
            ui.heading("🤖 AICHAT — Your AI Chat Agent");

            // Display messages
            for msg in &self.messages {
                ui.label(msg);
            }

            // Input area
            ui.horizontal(|ui| {
                ui.add(
                    egui::TextEdit::multiline(&mut self.user_input)
                        .desired_width(f32::INFINITY)
                        .hint_text("Type your message..."),
                );
                if ui.button("Send").clicked() && !self.user_input.is_empty() {
                    let input = self.user_input.clone();
                    self.messages.push(format!("👤 {}", input));

                    // Spawn async task to simulate AI response
                    let agent = self.agent.clone();
                    let messages = &mut self.messages;
                    tokio::spawn(async move {
                        match agent.send_message(&input).await {
                            Ok(response) => {
                                let mut msg_vec = messages.lock().await;
                                msg_vec.push(format!("🤖 {}", response));
                            }
                            Err(e) => {
                                let mut msg_vec = messages.lock().await;
                                msg_vec.push(format!("❌ Error: {}", e));
                            }
                        }
                    });
                    self.user_input.clear();
                }
            });

            // Auto-scroll to bottom
            if !self.messages.is_empty() {
                ui.scroll_area().auto_scroll_vertically(true).show(ui, |ui| {
                    for msg in &self.messages {
                        ui.label(msg);
                    }
                });
            }
        });
    }
}

fn main() -> Result<(), eframe::Error> {
    env_logger::init(); // optional: logging

    let options = eframe::NativeOptions {
        viewport: egui::ViewportBuilder::default().with_inner_size([800.0, 600.0]),
        default_theme: Theme::Dark,
        ..Default::default()
    };

    eframe::run_native(
        "AICHAT",
        options,
        Box::new(|_cc| Ok(Box::new(AICHTApp::default()))),
    )
}
```

## ✅ `ach_agent/src/lib.rs` — MAIN ENTRY POINT

```rust
pub mod config;
pub mod storage;
pub mod ollama_client;
pub mod openai_client;

use std::sync::Arc;
use tokio::sync::Mutex;

#[derive(Debug, Clone)]
pub struct ChatMessage {
    pub role: String,   // "user" or "assistant"
    pub content: String,
}

#[derive(Debug)]
pub struct AiChatAgent {
    history: Arc<Mutex<Vec<ChatMessage>>>,
    config: Arc<Mutex<config::Config>>,
    ollama_client: ollama_client::OllamaClient,
    openai_client: Option<openai_client::OpenAIClient>,
}

impl AiChatAgent {
    pub fn new() -> Self {
        let config = config::load_or_create_default().unwrap_or_else(|_| config::Config::default());
        let ollama_client = ollama_client::OllamaClient::new();
        let openai_client = if !config.openai_api_key.is_empty() {
            Some(openai_client::OpenAIClient::new(&config.openai_api_key))
        } else {
            None
        };

        Self {
            history: Arc::new(Mutex::new(Vec::new())),
            config: Arc::new(Mutex::new(config)),
            ollama_client,
            openai_client,
        }
    }

    pub async fn send_message(&self, user_input: &str) -> Result<String, Box<dyn std::error::Error>> {
        let mut history = self.history.lock().await;
        history.push(ChatMessage {
            role: "user".to_string(),
            content: user_input.to_string(),
        });

        let config = self.config.lock().await.clone();
        let response = match config.backend {
            config::Backend::Ollama => {
                self.ollama_client.chat(user_input, &history).await?
            }
            config::Backend::OpenAI => {
                if let Some(client) = &self.openai_client {
                    client.chat(user_input, &history).await?
                } else {
                    return Err("OpenAI backend selected but no API key provided".into());
                }
            }
        };

        history.push(ChatMessage {
            role: "assistant".to_string(),
            content: response.clone(),
        });

        Ok(response)
    }

    pub async fn get_history(&self) -> Vec<ChatMessage> {
        let history = self.history.lock().await;
        history.clone()
    }

    pub async fn clear_history(&self) {
        let mut history = self.history.lock().await;
        history.clear();
    }

    pub async fn save_history(&self) -> Result<(), Box<dyn std::error::Error>> {
        let history = self.get_history().await;
        storage::save_history(&history).await?;
        Ok(())
    }

    pub async fn load_history(&self) -> Result<(), Box<dyn std::error::Error>> {
        let loaded = storage::load_history().await?;
        let mut history = self.history.lock().await;
        *history = loaded;
        Ok(())
    }

    pub async fn get_config(&self) -> config::Config {
        self.config.lock().await.clone()
    }

    pub async fn update_config(&self, new_config: config::Config) {
        let mut config = self.config.lock().await;
        *config = new_config;
        config.save().unwrap_or_else(|e| eprintln!("Failed to save config: {}", e));
    }
}
```

```rust
use serde::{Deserialize, Serialize};
use std::fs;
use std::path::PathBuf;
use dirs;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Config {
    pub backend: Backend,
    pub ollama_model: String,
    pub openai_api_key: String,
    pub openai_model: String,
    pub temperature: f32,
    pub max_tokens: u32,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum Backend {
    Ollama,
    OpenAI,
}

impl Default for Config {
    fn default() -> Self {
        Self {
            backend: Backend::Ollama,
            ollama_model: "llama3".to_string(),
            openai_api_key: "".to_string(),
            openai_model: "gpt-4o".to_string(),
            temperature: 0.7,
            max_tokens: 2048,
        }
    }
}

impl Config {
    pub fn save(&self) -> Result<(), Box<dyn std::error::Error>> {
        let config_dir = dirs::config_dir().ok_or("Could not find config directory")?;
        let config_path = config_dir.join("aichat").join("config.toml");

        fs::create_dir_all(config_path.parent().unwrap())?;
        let toml_str = toml::to_string(self)?;
        fs::write(config_path, toml_str)?;

        Ok(())
    }
}

pub fn load_or_create_default() -> Result<Config, Box<dyn std::error::Error>> {
    let config_dir = dirs::config_dir().ok_or("Could not find config directory")?;
    let config_path = config_dir.join("aichat").join("config.toml");

    if config_path.exists() {
        let toml_str = fs::read_to_string(config_path)?;
        let config: Config = toml::from_str(&toml_str)?;
        Ok(config)
    } else {
        let default = Config::default();
        default.save()?;
        Ok(default)
    }
}
```

```rust
use serde::{Deserialize, Serialize};
use std::fs;
use std::path::PathBuf;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SavedChatHistory {
    pub messages: Vec<ChatMessage>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ChatMessage {
    pub role: String,
    pub content: String,
}

pub async fn save_history(history: &[ChatMessage]) -> Result<(), Box<dyn std::error::Error>> {
    let data_dir = dirs::data_dir().ok_or("Could not find data directory")?;
    let history_path = data_dir.join("aichat").join("history.json");

    fs::create_dir_all(history_path.parent().unwrap())?;

    let saved = SavedChatHistory {
        messages: history.to_vec(),
    };

    let json_str = serde_json::to_string_pretty(&saved)?;
    fs::write(history_path, json_str)?;

    Ok(())
}

pub async fn load_history() -> Result<Vec<ChatMessage>, Box<dyn std::error::Error>> {
    let data_dir = dirs::data_dir().ok_or("Could not find data directory")?;
    let history_path = data_dir.join("aichat").join("history.json");

    if !history_path.exists() {
        return Ok(Vec::new());
    }

    let json_str = fs::read_to_string(history_path)?;
    let saved: SavedChatHistory = serde_json::from_str(&json_str)?;
    Ok(saved.messages)
}
```

## ✅ `ach_agent/src/ollama_client.rs`

```rust
use reqwest;
use serde::{Deserialize, Serialize};
use std::time::Duration;

#[derive(Serialize)]
struct OllamaChatRequest {
    model: String,
    messages: Vec<OllamaMessage>,
    stream: bool,
}

#[derive(Serialize)]
struct OllamaMessage {
    role: String,
    content: String,
}

#[derive(Deserialize)]
struct OllamaChatResponse {
    message: OllamaMessage,
    done: bool,
}

pub struct OllamaClient {
    client: reqwest::Client,
}

impl OllamaClient {
    pub fn new() -> Self {
        Self {
            client: reqwest::Client::builder()
                .timeout(Duration::from_secs(120))
                .build()
                .unwrap(),
        }
    }

    pub async fn chat(
        &self,
        user_input: &str,
        history: &[super::ChatMessage],
    ) -> Result<String, Box<dyn std::error::Error>> {
        let url = "http://localhost:11434/api/chat";

        let messages: Vec<OllamaMessage> = history
            .iter()
            .map(|msg| OllamaMessage {
                role: msg.role.clone(),
                content: msg.content.clone(),
            })
            .collect();

        let request = OllamaChatRequest {
            model: "llama3".to_string(), // configurable later
            messages,
            stream: true,
        };

        let mut stream = self.client.post(url)
            .json(&request)
            .send()
            .await?
            .bytes_stream();

        let mut full_response = String::new();
        while let Some(chunk) = stream.next().await {
            let chunk = chunk?;
            let str_chunk = String::from_utf8_lossy(&chunk);
            for line in str_chunk.lines() {
                if line.starts_with("data: ") {
                    let json_str = &line["data: ".len()..];
                    if json_str == "[DONE]" {
                        break;
                    }
                    if let Ok(response) = serde_json::from_str::<OllamaChatResponse>(json_str) {
                        full_response.push_str(&response.message.content);
                    }
                }
            }
        }

        Ok(full_response)
    }
}
```
